19  /* This is an ItemManager that takes a list of labels
20  and a list of corresponding callback pointers */
21  class ListItemManager final: public ItemManager {
22  public:
23 
24  /* one item consisting of one label and one callback */
25  struct Item {
26  char label[8]; // non-null-terminated string
27  void (*callback)();
28 
29  Item(): label{'\0'}, callback(nullptr){}
30  };
31 
32  /* function that helps to read continuously stored labels and callbacks */
33  /* consider [size] Items that are stored the following way...
34  [labels] is a string with all item labels separated with '\n', ending with \0,
35  [callbacks] is a pointer to an array of function pointers corresponding to these labels
36  [index] is the index of a specific Item.
37  Then this function picks out the Item at position [index], i.e.
38  grab the label as well as callback, put them into an Item object an return it.
39  If given [index] >= [size], a standard Item will be returned (empty string + nullptr)
40  */
41  static Item item_list_decoder(uint8_t index, uint8_t size, const char* labels, void (* const *callbacks)());
42 
43  /* example how to use item_list_decoder */
44  inline static Item item_list_array_example(uint8_t index){
45  const uint8_t SIZE = 4;
46  const char labels[] =
47 R"xxx(item 1
48 item 2
49 item 3
50 item 4)xxx";
51  void (* const callbacks[SIZE])() = {nullptr, nullptr,nullptr,nullptr};
52  return item_list_decoder(index,SIZE,labels,&callbacks[0]);
53  }
54 
55  private:
56 
57  /* Function pointer to a function returning the Item at given index. */
58  /* This way extern functions provides all the data needed here. */
59  Item (*item_array)(uint8_t index);
60 
61  /* size of list (no cancel item is encountered) */
62  uint8_t size;
63 
64  protected:
65 
66  /* provide item label for superclass */
67  virtual void item_label_virt(char* string_8_bytes) override {
68  if (item_array != nullptr) return hardware::copyString(string_8_bytes,&(item_array(position).label[0]),8,false);//this copying should be done in the item array
69  }
70 
71  /* provide callback function pointer for superclass */
72  virtual void run_item_procedure_virt() override {
73  if (item_array != nullptr) return item_array(position).callback();
74  }
75 
76  /* provide size for superclass */
77  virtual uint16_t size_without_cancel_item() override {
78  return size;
79  }
80 
81  public:
82 
83  /* minimalistic std c-tor. use init() for initialization */
84  ListItemManager() : ItemManager() , item_array(nullptr), size(0){}
85 
86  /* init the item manager, set array function, list size, cancel event callback, and position in list where to start */
87  virtual void init( Item (*item_array_function)(uint8_t), uint8_t size_without_cancel_item, void (*cancel_procedure)() = nullptr, int16_t std_position = 0){
88  this->item_array = item_array_function;
89  this->size = size_without_cancel_item;
90  ItemManager::init(cancel_procedure, std_position);
91  }
92 
93  /* reset item manager to right after construction.
94  run init() next. */
95  void reset(){
96  size = 0;
97  item_array = nullptr;
98  }
99  };
100 }
