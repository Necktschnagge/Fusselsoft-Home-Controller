79  inline void init(bool silence = false){//for init
80  // IO register B for the lights:
81  PORTB &= 0b11111000;
82  DDRB |= 0b00000111;// LATCH BIT ::: CLOCK BIT ::: DATA BIT
83 
84  pushLineVisible(0x0000);
85  if (!silence){
86  hardware::delay(1000);

88  uint16_t sig = 0;
89  for(uint8_t count = 1; count <= 33; ++count){
90  pushLineVisible(sig);
91  hardware::delay(100);
92  sig = (sig << 1) + !(sig & 0x8000);

94  hardware::delay(1000);
