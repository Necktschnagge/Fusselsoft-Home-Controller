50  /* it may be decreased from extern when calling the fade_out() method */

52  /* the fade_out() is not need to be called */

274  template<typename Metric, uint8_t channels>
275  bool ChannellingMovementRecognizer<Metric,channels>::Channel::accumulate(const Metric& new_value, Configuration* config, Event& event){
276  if ( (is_valid())
277  &&
278  (current_value - config->epsilon <= new_value) && (new_value <= current_value + config->epsilon)
279  ){
280  current_value = new_value;
281  event[0] = latest_zone;
282  event[1] = get_zone(new_value,config);
283  if (event[1] != NO_ZONE) latest_zone = event[1];
284  reset_fade();
285  unworse();
286  better(config);
287  return true;
288  }
289  return false;
290  }

310  uint8_t ChannellingMovementRecognizer<Metric,channels>::get_zone(const Metric& distance, Configuration* config){
311  Zone* zptr = config->zones;
312  for (uint8_t zone = 0; zone < config->count_zones; ++zone){
313  if ((zptr[zone].lower_border <= distance) && (distance <= zptr[zone].upper_border)){
314  return zone;

320  template<typename Metric, uint8_t channels>
321  void ChannellingMovementRecognizer<Metric,channels>::smart_sort(uint8_t channel){
322  uint8_t i_channel {channel};
323  // shuffle to the front:
324  while (1){
325  if (channel == 0) goto part2; // we are at the front
326  if (!(ch[channel] < ch[channel-1])) goto part2; // we don't need to reorder
327  // we need to sort, i.e. push this channel left:
328  // swap with previous channel:
329  Channel swap;
330  swap = ch[channel-1];
331  ch[channel-1] = ch[channel];
332  ch[channel] = swap;
333  --channel;
334  // <<< better bytewise swapping!! add such function in hardware.h of fussel fframework
335  }
336  part2:
337  //shuffle to the back:
338  while (1){
339  if (i_channel == channels-1) return; // reached last channel
340  if (!( ch[i_channel] > ch[i_channel+1] )) return; // reordering not necessary
341  // this channel is greater than next one: push it to the right:
342  Channel swap;
343  swap = ch[i_channel+1];
344  ch[i_channel+1] = ch[i_channel];
345  ch[i_channel] = swap;
346  ++i_channel;
347  }
348  }

350  template<typename Metric, uint8_t channels>
351  void ChannellingMovementRecognizer<Metric,channels>::put(const Metric& value, uint8_t& from, uint8_t& to){
352  Event my_event;
353  for (uint8_t i = 0; i<channels; ++i){
354  ch[i].worse(config);
355  }
356  for (uint8_t i = 0; i<channels; ++i){
357  if (ch[i].accumulate(value, config, my_event)){
358  // one Channel matched:
359  from = my_event.from;
360  to = my_event.to;
361  smart_sort(i);
362  return;
363  }
364  }
365  // no Channel matched:
366  // ch[channels-1].~Channel(); // destroy worst channel // optional because we override it
367  ch[channels-1] = Channel(value,config); // replace worst channel
368  from = NO_ZONE;
369  to = ch[channels-1].get_latest_zone();
370  smart_sort(channels-1);
371  }

373  template<typename Metric, uint8_t channels>
374  void ChannellingMovementRecognizer<Metric,channels>::fade_out(uint8_t amount){
375  // when fading out, the only change with effects to array order is
376  // that a channel can become invalid and has to be toggled to the end.
377  // When beginning at the end of the array,
378  // we don't risk that a Channel will be left out because of sorting / toggelling.
379  for (uint8_t c = channels; c !=0;){
380  --c;
381  if (ch[c].fade_out(amount)) smart_sort(c);
382  // when faded out we might need to refresh array order (shuffle to back)
383  }
384  }
