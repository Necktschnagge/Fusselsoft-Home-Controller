95  inline static base_type addition(base_type summand0, base_type summand1){
96  // restructure as soon as if constexpr is supported.
97  if (summand0 == base_type_constants::OUT_OF_RANGE) return OUT_OF_RANGE_IS_ABSORBING ? base_type_constants::OUT_OF_RANGE : summand1;
98  if (summand1 == base_type_constants::OUT_OF_RANGE) return OUT_OF_RANGE_IS_ABSORBING ? base_type_constants::OUT_OF_RANGE : summand0;
99  const base_type raw_sum{ summand0 + summand1 };
100  if (OVERFLOW_RESULTS_IN_OUT_OF_RANGE){

102  } else {
103  return raw_sum % RANGE;
104  }
