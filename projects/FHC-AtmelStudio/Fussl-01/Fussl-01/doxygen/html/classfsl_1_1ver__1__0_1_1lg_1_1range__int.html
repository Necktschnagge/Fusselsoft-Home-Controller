template<typename _base_type, _base_type _RANGE, bool _OVERFLOW_RESULTS_IN_OUT_OF_RANGE = true, bool _OUT_OF_RANGE_IS_ABSORBING = true>
class fsl::ver_1_0::lg::range_int< _base_type, _RANGE, _OVERFLOW_RESULTS_IN_OUT_OF_RANGE, _OUT_OF_RANGE_IS_ABSORBING >

Type for Integers in a range {base_type_constants::ZERO, ... , base_type_constants::MAX}.

Stores a number x in {base_type_constants::ZERO, ... , base_type_constants::MAX} or OUT_OF_RANGE.
