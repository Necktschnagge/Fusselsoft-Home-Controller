53  template <uint8_t x_size, uint8_t y_size>
54  class matrix_keyboard {
55 
56  static_assert(static_cast<uint16_t>(x_size) + static_cast<uint16_t>(y_size) < 255, "matrix size too large, x_size + y_size < 256 must hold");
57 
58  private:
59 
60  /* points to first GPIOPin of x-row/col pins, y pins follow immediately
61  reserved pins are {x_array_begin, ... , x_array_begin + x_size + y_size -1 }
62  */
63  fsl::hw::gpio_pin _x_array_begin;
64 
65  constexpr fsl::hw::gpio_pin array_begin(){ return _x_array_begin; }
66  constexpr fsl::hw::gpio_pin array_end(){ return _x_array_begin + count_GPIOs(); }
67  constexpr fsl::hw::gpio_pin x_array_begin(){ return _x_array_begin; }
68  constexpr fsl::hw::gpio_pin x_array_end(){ return _x_array_begin + x_size; }
69  constexpr fsl::hw::gpio_pin y_array_begin(){ return _x_array_begin + x_size; }
70  constexpr fsl::hw::gpio_pin y_array_end(){ return _x_array_begin + count_GPIOs(); }
71 
72  public:
73 
74  /* the number of occupied GPIO pins */
75  static constexpr uint8_t count_GPIOs(){ return x_size + y_size; }
76  /* the number of (theoretical) keys on the matrix keyboard */
77  static constexpr uint16_t key_count(){ return static_cast<uint16_t>(x_size) * y_size; }
78  /* the past-the-end iterator for the used GPIO pins*/
79  constexpr fsl::hw::gpio_pin GPIO_end(){ return array_end(); } // <<< is constexpr allowed here???

81  /* enum class for number of pressed keys */
82  enum class key_down_count : uint8_t {
83  NONE_PRESSED = 0,
84  ONE_PRESSED = 1,
85  AMBIGUOUS = 2
86  };
87 
88  static constexpr key_down_count NONE_PRESSED {key_down_count::NONE_PRESSED};
89  static constexpr key_down_count ONE_PRESSED {key_down_count::ONE_PRESSED};
90  static constexpr key_down_count AMBIGUOUS {key_down_count::AMBIGUOUS};
91 
92  /* class describing the state of the key matrix at one point in time */
93  // <<< this struct is independent of all tmeplate params of nesting class. so might be moved outerwards?? or replace the uint8_t with some range_int???
94  // <<<<< usingh range int might be unnecessary overhead? ... or could otherwise we could remove the key_down_count conter; (????)
95  struct key_state {
96  key_down_count counter; // "amount" of pressed keys: NONE / ONE / AMBIGUOUS
97  uint8_t x; // x position of the ONE key that was pressed (if so)
98  uint8_t y; // y position of the ONE key that was pressed (if so)
99 
100  key_state() : counter(key_down_count::NONE_PRESSED), x(0), y(0) {}
101 
102  /* increments counter, ambiguous will be fixed by this operation */
103  /* NONE_PRESSED |-> ONE_PRESSED
104  ONE_PRESSED |-> AMBIGUOUS
105  AMBIGUOUS |-> AMBIGUOUS
106  */
107  inline void inc_counter(){
108  uint8_t x = static_cast<uint8_t>(counter);
109  counter = static_cast<key_down_count>( (2 + 3 * x - x * x) / 2 );
110  // 0 -> 1, 1 -> 2, 2 -> 2
111  }
112 
113  /* conversion to bool: evaluates to true if and only if one and only one key is pressed */
114  inline operator bool(){ return counter==key_down_count::ONE_PRESSED; }
115 
116  /* operator (): returns the key code for the key that was pressed */
117  inline uint16_t operator()(){ return static_cast<uint16_t>(x) + static_cast<uint16_t>(x_size) * y; }
118  };
119 
120  /* x_array_begin points to first GPIOPin of x-row/col pins, y pins follow immediately
121  reserved pins are {x_array_begin, ... , x_array_begin + x_size + y_size -1 }
122  past-the-end iterator can be accessed via GPIO_end()
123  make sure this class is exclusively using these pins
124  */
125  inline matrix_keyboard(fsl::hw::gpio_pin x_array_begin) : _x_array_begin(x_array_begin) {
126  for(fsl::hw::gpio_pin iter = array_begin(); iter != array_end(); ++ iter){
127  iter.set_as_input(); // get a safe state first
128  iter.set_pull_up(true); // activate pull-up
129  }
130  }
131 
132  /* read the current key state of matrix_keyboard */
133  key_state operator()(){
134  key_state result;
135  for(fsl::hw::gpio_pin iter = x_array_begin(); iter != x_array_end() ; ++ iter){
136  iter.write_PORT(false); // i.e. deactivate pullup, goal: output 0V
137  iter.set_as_output(); // output 0 on pin iter (x-panel)
138  for(fsl::hw::gpio_pin jter = y_array_begin(); jter != y_array_end(); ++jter){
139  if (jter.read_PIN() == false){
140  result.inc_counter();
141  result.x = iter - x_array_begin();
142  result.y = jter - y_array_end();
143  }
144  }
145  iter.set_as_input();
146  iter.write_PORT(true); // enable intern pull up
147  }
148  return result;
149  }
150  };

153  }
