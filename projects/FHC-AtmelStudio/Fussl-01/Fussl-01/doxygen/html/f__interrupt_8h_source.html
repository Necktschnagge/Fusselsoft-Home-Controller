25  /* The same behaviour as standard_atomic but sizeof(lite_atomic) == 1 instead of 2 like for standard_atomic */
26  class lite_atomic {
27  fsl::lg::single_flags flags;
28  fsl::lg::single_flags::flag_id ACTIVE{ 0 };
29  fsl::lg::single_flags::flag_id STORED_I_FLAG{ 1 };
30 
31  inline static bool sreg_I(){ return (SREG & (1<<7)); }
32 
33  /* copy global interrupt enable flag and deactivate interrupts */
34  inline void store(){ flags.set(STORED_I_FLAG, sreg_I()); cli(); }
35  /* restore gloabl interrupt enable flag to register sreg iff active() */
36  inline void restore(){ if (active()) { if (flags.get(STORED_I_FLAG)) sei(); else cli(); } }
37 
38  public:
39 
40  /* create lite_atomic and immediately go into critical section / go into active state */
41  inline lite_atomic() { fsl::lg::single_flags::memory_t f_mem{ static_cast<uint8_t>(1 + 2 * sreg_I()) }; flags = fsl::lg::single_flags(f_mem); cli(); }
42  /* create lite_atomic, go into critical section / active state iff active. */
43  inline lite_atomic(bool active) { fsl::lg::single_flags::memory_t f_mem{ active }; flags = fsl::lg::single_flags(f_mem); if (active) store(); }
44  /* leave critical section / active state if in this state */
45  inline ~lite_atomic() { restore(); }

51  inline lite_atomic(lite_atomic&& obj) : flags(obj.flags) { obj.flags.set_false(ACTIVE); }
52  /* leave the own critical section if necessary, i.e. if active, then take state of obj and let obj become an inactive lite_atomic */
53  inline lite_atomic& operator = (lite_atomic&& obj) { restore(); flags = obj.flags; obj.flags.set_false(ACTIVE); return *this; }

68  /* A standard_atomic is a more advanced variant of a simple_atomic. You may see the description of simple_atomic to get familiar with the concept.
69  A standard_atomic can on the one hand disable interrupts and save the MCUs interrupt configuration register SREG in a copy
70  and on the other hand restore the MCUs interrupt configuration register from the copy saved.
71  A standard_atomic (in difference to simple_atomic) may block interrupts and unblock them again as often as desired.
72  */
73  class standard_atomic {
74 
75  bool _active;
76  uint8_t sreg;
77 
78  /* copy interrupt register and deactivate interrupts */
79  inline void store(){ sreg = SREG; cli(); }
80  /* restore the register copy to register iff active() */
81  inline void restore(){ if (_active) SREG = sreg; }
82 
83  public:
84 
85  /* create standard_atomic and immediately go into critical section / go into active state */
86  inline standard_atomic() : _active(true), sreg(SREG) { cli(); }
87  /* create standard_atomic, go into critical section / active state iff active. */
88  inline standard_atomic(bool active) : _active(active) { if (active) store(); }
89  /* leave critical section / active state if in this state */
90  inline ~standard_atomic() { restore(); }
91 
92  standard_atomic(const standard_atomic&) = delete;
93  standard_atomic& operator = (const standard_atomic&) = delete;
94 
95  /* take state of obj and let obj become an inactive standard_atomic */
96  inline standard_atomic(standard_atomic&& obj) : _active(obj._active), sreg(obj.sreg) { obj._active = false; }
97  /* leave the own critical section if necessary, i.e. if active, then take state of obj and let obj become an inactive standard_atomic */
98  inline standard_atomic& operator = (standard_atomic&& obj) { restore(); _active = obj._active; sreg = obj.sreg; obj._active = false; return *this; }
99 
100  /* go into critical section (iff not already in) */
101  inline void open(){ if (!_active) { store(); _active = true; } }
102  /* go into critical section (iff not already in) */
103  inline void p(){ open(); }
104  /* leave critical section (iff not already inactive) */
105  inline void close(){ restore(); _active = false; }
106  /* leave critical section (iff not already inactive) */
107  inline void v(){ close(); }
108 
109  /* returns true iff standard_atomic is in its critical section */
110  inline bool active(){ return _active; }
111  };
